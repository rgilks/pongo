8use core::sync::atomic::{AtomicUsize, Ordering};
9
10/// A bit mask used to signal the `AtomicBorrow` has an active mutable borrow.
11const UNIQUE_BIT: usize = !(usize::max_value() >> 1);
12
13const COUNTER_MASK: usize = usize::max_value() >> 1;
14
15/// An atomic integer used to dynamicaly enforce borrowing rules
16///
17/// The most significant bit is used to track mutable borrow, and the rest is a
18/// counter for immutable borrows.
19///
20/// It has four possible states:
21///  - `0b00000000...` the counter isn't mut borrowed, and ready for borrowing
22///  - `0b0_______...` the counter isn't mut borrowed, and currently borrowed
23///  - `0b10000000...` the counter is mut borrowed
24///  - `0b1_______...` the counter is mut borrowed, and some other thread is trying to borrow
25pub struct AtomicBorrow(AtomicUsize);
26
27impl AtomicBorrow {
28    pub const fn new() -> Self {
29        Self(AtomicUsize::new(0))
30    }
31
32    pub fn borrow(&self) -> bool {
33        // Add one to the borrow counter
34        let prev_value = self.0.fetch_add(1, Ordering::Acquire);
35
36        // If the previous counter had all of the immutable borrow bits set,
37        // the immutable borrow counter overflowed.
38        if prev_value & COUNTER_MASK == COUNTER_MASK {
39            core::panic!("immutable borrow counter overflowed")
40        }
41
42        // If the mutable borrow bit is set, immutable borrow can't occur. Roll back.
43        if prev_value & UNIQUE_BIT != 0 {
44            self.0.fetch_sub(1, Ordering::Release);
45            false
46        } else {
47            true
48        }
49    }
50
51    pub fn borrow_mut(&self) -> bool {
52        self.0
53            .compare_exchange(0, UNIQUE_BIT, Ordering::Acquire, Ordering::Relaxed)
54            .is_ok()
55    }
56
57    pub fn release(&self) {
58        let value = self.0.fetch_sub(1, Ordering::Release);
59        debug_assert!(value != 0, "unbalanced release");
60        debug_assert!(value & UNIQUE_BIT == 0, "shared release of unique borrow");
61    }
62
63    pub fn release_mut(&self) {
64        let value = self.0.fetch_and(!UNIQUE_BIT, Ordering::Release);
65        debug_assert_ne!(value & UNIQUE_BIT, 0, "unique release of shared borrow");
66    }
67}

69#[cfg(test)]
70mod tests {
71    use super::*;
72
73    #[test]
74    #[should_panic(expected = "immutable borrow counter overflowed")]
75    fn test_borrow_counter_overflow() {
76        let counter = AtomicBorrow(AtomicUsize::new(COUNTER_MASK));
77        counter.borrow();
78    }
79
80    #[test]
81    #[should_panic(expected = "immutable borrow counter overflowed")]
82    fn test_mut_borrow_counter_overflow() {
83        let counter = AtomicBorrow(AtomicUsize::new(COUNTER_MASK | UNIQUE_BIT));
84        counter.borrow();
85    }
86
87    #[test]
88    fn test_borrow() {
89        let counter = AtomicBorrow::new();
90        assert!(counter.borrow());
91        assert!(counter.borrow());
92        assert!(!counter.borrow_mut());
93        counter.release();
94        counter.release();
95
96        assert!(counter.borrow_mut());
97        assert!(!counter.borrow());
98        counter.release_mut();
99        assert!(counter.borrow());
100    }
